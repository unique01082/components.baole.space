## 💡 Usage Examples

### Import Components
```typescript
import {
  Button, Card, Input, Dialog, Tabs,
  Alert, Progress, Avatar, Badge,
  // ... 78 more components
} from '@baole/ui'
```

### Basic Form
```tsx
<Form onSubmit={handleSubmit}>
  <FormItem label="Email" required>
    <Input type="email" />
  </FormItem>
  <Button variant="gradient">Submit</Button>
</Form>
```

### Dialog Pattern
```tsx
<Dialog>
  <DialogTrigger asChild>
    <Button>Open</Button>
  </DialogTrigger>
  <DialogContent>
    <DialogHeader>
      <DialogTitle>Title</DialogTitle>
    </DialogHeader>
    <p>Content</p>
  </DialogContent>
</Dialog>
```
